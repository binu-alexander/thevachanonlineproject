36
1 וַיֹּ֥סֶף אֱלִיה֗וּא וַיֹּאמַֽר׃ 2 כַּתַּר־לִ֣י זְ֭עֵיר וַאֲחַוֶּ֑ךָּ כִּ֤י ע֖וֹד לֶאֱל֣וֹהַּ מִלִּֽים׃ 3 אֶשָּׂ֣א דֵ֭עִי לְמֵרָח֑וֹק וּ֝לְפֹעֲלִ֗י אֶֽתֵּֽן־צֶֽדֶק׃ 4 כִּֽי־אָ֭מְנָם לֹא־שֶׁ֣קֶר מִלָּ֑י תְּמִ֖ים דֵּע֣וֹת עִמָּֽךְ׃ 5 הֶן־אֵ֣ל כַּ֭בִּיר וְלֹ֣א יִמְאָ֑ס כַּ֝בִּ֗יר כֹּ֣חַֽ לֵֽב׃ 6 לֹא־יְחַיֶּ֥ה רָשָׁ֑ע וּמִשְׁפַּ֖ט עֲנִיִּ֣ים יִתֵּֽן׃ 7 לֹֽא־יִגְרַ֥ע מִצַּדִּ֗יק עֵ֫ינָ֥יו וְאֶת־מְלָכִ֥ים לַכִּסֵּ֑א וַיֹּשִׁיבֵ֥ם לָ֝נֶ֗צַח וַיִּגְבָּֽהוּ׃ 8 וְאִם־אֲסוּרִ֥ים בַּזִּקִּ֑ים יִ֝לָּכְד֗וּן בְּחַבְלֵי־עֹֽנִי׃ 9 וַיַּגֵּ֣ד לָהֶ֣ם פָּעֳלָ֑ם וּ֝פִשְׁעֵיהֶ֗ם כִּ֣י יִתְגַּבָּֽרוּ׃ 10 וַיִּ֣גֶל אָ֭זְנָם לַמּוּסָ֑ר וַ֝יֹּ֗אמֶר כִּֽי־יְשֻׁב֥וּן מֵאָֽוֶן׃ 11 אִֽם־יִשְׁמְע֗וּ וְֽיַ֫עֲבֹ֥דוּ יְכַלּ֣וּ יְמֵיהֶ֣ם בַּטּ֑וֹב וּ֝שְׁנֵיהֶ֗ם בַּנְּעִימִֽים׃ 12 וְאִם־לֹ֣א יִ֭שְׁמְעוּ בְּשֶׁ֣לַח יַעֲבֹ֑רוּ וְ֝יִגְוְע֗וּ כִּבְלִי־דָֽעַת׃ 13 וְֽחַנְפֵי־לֵ֭ב יָשִׂ֣ימוּ אָ֑ף לֹ֥א יְ֝שַׁוְּע֗וּ כִּ֣י אֲסָרָֽם׃ 14 תָּמֹ֣ת בַּנֹּ֣עַר נַפְשָׁ֑ם וְ֝חַיָּתָ֗ם בַּקְּדֵשִֽׁים׃ 15 יְחַלֵּ֣ץ עָנִ֣י בְעָנְי֑וֹ וְיִ֖גֶל בַּלַּ֣חַץ אָזְנָֽם׃ 16 וְאַ֤ף הֲסִיתְךָ֨ ׀ מִפִּי־צָ֗ר רַ֭חַב לֹא־מוּצָ֣ק תַּחְתֶּ֑יהָ וְנַ֥חַת שֻׁ֝לְחָנְךָ֗ מָ֣לֵא דָֽשֶׁן׃ 17 וְדִין־רָשָׁ֥ע מָלֵ֑אתָ דִּ֖ין וּמִשְׁפָּ֣ט יִתְמֹֽכוּ׃ 18 כִּֽי־חֵ֭מָה פֶּן־יְסִֽיתְךָ֣ בְסָ֑פֶק וְרָב־כֹּ֝֗פֶר אַל־יַטֶּֽךָּ׃ 19 הֲיַעֲרֹ֣ךְ שׁ֭וּעֲךָ לֹ֣א בְצָ֑ר וְ֝כֹ֗ל מַאֲמַצֵּי־כֹֽחַ׃ 20 אַל־תִּשְׁאַ֥ף הַלָּ֑יְלָה לַעֲל֖וֹת עַמִּ֣ים תַּחְתָּֽם׃ 21 הִ֭שָּׁמֶר אַל־תֵּ֣פֶן אֶל־אָ֑וֶן כִּֽי־עַל־זֶ֝֗ה בָּחַ֥רְתָּ מֵעֹֽנִי׃ 22 הֶן־אֵ֭ל יַשְׂגִּ֣יב בְּכֹח֑וֹ מִ֖י כָמֹ֣הוּ מוֹרֶֽה׃ 23 מִֽי־פָקַ֣ד עָלָ֣יו דַּרְכּ֑וֹ וּמִֽי־אָ֝מַ֗ר פָּעַ֥לְתָּ עַוְלָֽה׃ 24 זְ֭כֹר כִּֽי־תַשְׂגִּ֣יא פָעֳל֑וֹ אֲשֶׁ֖ר שֹׁרְר֣וּ אֲנָשִֽׁים׃ 25 כָּל־אָדָ֥ם חָֽזוּ־ב֑וֹ אֱ֝נ֗וֹשׁ יַבִּ֥יט מֵרָחֽוֹק׃ 26 הֶן־אֵ֣ל שַׂ֭גִּיא וְלֹ֣א נֵדָ֑ע מִסְפַּ֖ר שָׁנָ֣יו וְלֹא־חֵֽקֶר׃ 27 כִּ֭י יְגָרַ֣ע נִטְפֵי־מָ֑יִם יָזֹ֖קּוּ מָטָ֣ר לְאֵדֽוֹ׃ 28 אֲשֶֽׁר־יִזְּל֥וּ שְׁחָקִ֑ים יִ֝רְעֲפ֗וּ עֲלֵ֤י ׀ אָדָ֬ם רָֽב׃ 29 אַ֣ף אִם־יָ֭בִין מִפְרְשֵׂי־עָ֑ב תְּ֝שֻׁא֗וֹת סֻכָּתֽוֹ׃ 30 הֵן־פָּרַ֣שׂ עָלָ֣יו אוֹר֑וֹ וְשָׁרְשֵׁ֖י הַיָּ֣ם כִּסָּֽה׃ 31 כִּי־בָ֭ם יָדִ֣ין עַמִּ֑ים יִֽתֶּן־אֹ֥כֶל לְמַכְבִּֽיר׃ 32 עַל־כַּפַּ֥יִם כִּסָּה־א֑וֹר וַיְצַ֖ו עָלֶ֣יהָ בְמַפְגִּֽיעַ׃ 33 יַגִּ֣יד עָלָ֣יו רֵע֑וֹ מִ֝קְנֶ֗ה אַ֣ף עַל־עוֹלֶֽה׃
