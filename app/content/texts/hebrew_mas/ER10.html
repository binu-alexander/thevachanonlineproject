לֹֽא־שָׁתָ֔ה כִּ֥י מִתְאַבֵּ֖ל עַל־מַ֥עַל הַגֹּולָֽה׃ ס 7 וַיַּעֲבִ֨ירוּ קֹ֜ול בִּיהוּדָ֣ה וִירֽוּשָׁלַ֗͏ִם לְכֹל֙ בְּנֵ֣י הַגֹּולָ֔ה לְהִקָּבֵ֖ץ יְרוּשָׁלָֽ͏ִם׃ 8 וְכֹל֩ אֲשֶׁ֨ר לֹֽא־יָבֹ֜וא 13 אֲבָ֞ל הָעָ֥ם רָב֙ וְהָעֵ֣ת גְּשָׁמִ֔ים וְאֵ֥ין כֹּ֖חַ לַעֲמֹ֣וד בַּח֑וּץ וְהַמְּלָאכָ֗ה לֹֽא־לְיֹ֤ום אֶחָד֙ וְלֹ֣א לִשְׁנַ֔יִם כִּֽי־הִרְבִּ֥ינוּ לִפְשֹׁ֖עַ בַּדָּבָ֥ר הַזֶּֽה׃ 14 יַֽעֲמְדוּ־נָ֣א
