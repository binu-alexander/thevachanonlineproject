וְאַרְבַּ֥ע מֵאֹ֖ות שָׁנָֽה׃ 41 וַיְהִ֗י מִקֵּץ֙ שְׁלֹשִׁ֣ים שָׁנָ֔ה וְאַרְבַּ֥ע מֵאֹ֖ות שָׁנָ֑ה וַיְהִ֗י בְּעֶ֙צֶם֙ הַיֹּ֣ום הַזֶּ֔ה יָֽצְא֛וּ כָּל־צִבְאֹ֥ות יְהוָ֖ה מֵאֶ֥רֶץ מִצְרָֽיִם׃ 42 לֵ֣יל חֻקַּ֣ת הַפָּ֑סַח כָּל־בֶּן־נֵכָ֖ר לֹא־יֹ֥אכַל בֹּֽו׃ 44 וְכָל־עֶ֥בֶד אִ֖ישׁ מִקְנַת־כָּ֑סֶף וּמַלְתָּ֣ה אֹתֹ֔ו אָ֖ז יֹ֥אכַל בֹּֽו׃ 45 תֹּושָׁ֥ב וְשָׂכִ֖יר לֹא־יֹ֥אכַל־בֹּֽו׃ 46 בְּבַ֤יִת אֶחָד֙ יֵאָכֵ֔ל לֹא־תֹוצִ֧יא מִן־הַבַּ֛יִת מִן־הַבָּשָׂ֖ר ח֑וּצָה וְעֶ֖צֶם לֹ֥א תִשְׁבְּרוּ־בֹֽו׃ 47 כָּל־עֲדַ֥ת יִשְׂרָאֵ֖ל יַעֲשׂ֥וּ אֹתֹֽו׃ 48 וְכִֽי־יָג֨וּר אִתְּךָ֜ גֵּ֗ר וְעָ֣שָׂה פֶסַח֮ לַיהוָה֒
